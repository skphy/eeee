[docs]    def get_free_energy(self): #, Edft, S, temp):
        '''
        Units::  Edft : eV
                    S : eV/K
                 temp : K
        '''
        Edft = self.Edft
        S    = self.S
        temp = self.temp

        return {'free_energy': Edft - temp*S, 'units': 'eV'}
